-- 7-day rolling average PM2.5 by district SELECT district, measurement_date, pm25, AVG(pm25) OVER ( PARTITION BY district ORDER BY measurement_date ROWS BETWEEN 6 PRECEDING AND CURRENT ROW ) AS rolling_7day_avg FROM air_quality ORDER BY district, measurement_date;

Build the composite priority score with RANK()
Rank districts by both burden and asthma, then average the ranks to find highest combined need.
SELECT district, RANK() OVER (ORDER BY energy_burden_pct DESC) AS burden_rank, RANK() OVER (ORDER BY asthma_rate DESC) AS health_rank, (RANK() OVER (ORDER BY energy_burden_pct DESC) + RANK() OVER (ORDER BY asthma_rate DESC)) / 2.0 AS priority_score FROM combined_data ORDER BY priority_score;
Rewrite your messiest query as a CTE
Take Week 2's GROUP BY query and wrap it in a WITH clause — cleaner and reusable.
WITH borough_averages AS ( SELECT borough, AVG(asthma_rate) AS avg_asthma, AVG(energy_burden_pct) AS avg_burden FROM combined_data GROUP BY borough ) SELECT * FROM borough_averages WHERE avg_burden > (SELECT AVG(avg_burden) FROM borough_averages);

"I built a project analyzing the relationship between energy infrastructure and public health outcomes in New York City.
